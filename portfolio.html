// ระบบแจ้งเตือนเมื่อเซ็นเซอร์ตรวจจับความเร็ว (Random Simulation)
public class NotificationManager {

    private static final int SPEED_LIMIT     = 60;   // km/h
    private static final int DANGER_THRESHOLD = 80;   // km/h

    /**
     * ตรวจสอบความเร็วและส่งการแจ้งเตือนตามระดับความรุนแรง
     */
    public void processSensorReading(int speed, String zone) {

        if (speed > DANGER_THRESHOLD) {
            // 🚨 อันตราย — ส่งแจ้งเตือนด่วน
            sendAlert(
                "⚠️ ความเร็วเกินกำหนดอย่างมาก!",
                "ตรวจพบ " + speed + " km/h ณ " + zone,
                AlertLevel.DANGER
            );
        } else if (speed > SPEED_LIMIT) {
            // ⚡ คำเตือน
            sendAlert(
                "ความเร็วสูงกว่ากำหนด",
                speed + " km/h — " + zone,
                AlertLevel.WARNING
            );
        } else {
            // ✅ ปกติ — บันทึก log
            Log.d("SpeedGuard", "OK: " + speed + " km/h @ " + zone);
        }
    }

    private void sendAlert(String title, String body, AlertLevel level) {
        NotificationCompat.Builder builder = new NotificationCompat.Builder(context)
            .setSmallIcon(R.drawable.ic_speed)
            .setContentTitle(title)
            .setContentText(body)
            .setPriority(NotificationCompat.PRIORITY_HIGH)
            .setAutoCancel(true);

        NotificationManagerCompat.from(context).notify(notifId++, builder.build());
    }
}
